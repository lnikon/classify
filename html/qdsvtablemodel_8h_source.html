27 //a template-based matrix to store 2D data
28 template<class T>
29 class QDsvMatrix {
30  QList<QList<T>> m_data;
31  int m_rowCount;
32  int m_columnCount;
33 
34 public:
35  QDsvMatrix(int row = 0, int column = 0):
36  m_rowCount(row), m_columnCount(column)
37  {
38  if (rowCount() > 0 && columnCount() > 0) {
39  for (int i = 0; i < rowCount(); ++i) {
40  QList<T> rowData;
41  for (int j = 0; j < columnCount(); ++j) {
42  rowData.insert(j, T());
43  }
44  m_data.insert(i, rowData);
45  }
46  }
47  }
48 
49  T at(int row, int column) const {
50  if (row >= rowCount() || column >= columnCount())
51  return T();
52  QList<T> rowData = m_data.at(row);
53  return rowData.value(column);
54  }
55 
56  bool setValue(int row, int column, T value) {
57  if (column >= columnCount()) {
58  m_columnCount = column + 1;
59  for (int i = 0; i < rowCount(); ++i) {
60  QList<T> rowData = m_data.at(i);
61  while (rowData.size() < columnCount()) {
62  rowData.append(T());
63  }
64  m_data[i] = rowData;
65  }
66  }
67  if (row >= rowCount()) {
68  m_rowCount = row + 1;
69  while (m_data.size() < rowCount()) {
70  QList<T> rowData;
71  while (rowData.size() < columnCount()) {
72  rowData.append(T());
73  }
74  m_data.append(rowData);
75  }
76  }
77 
78  m_data[row][column] = value;
79  return true;
80  }
81 
82  int rowCount() const {
83  return m_rowCount;
84  }
85 
86  int columnCount() const {
87  return m_columnCount;
88  }
89 
90  void clear() {
91  m_rowCount = 0;
92  m_columnCount = 0;
93  m_data.clear();
94  }
95 
96  void append(const QList<T> &value) {
97  if (value.size() > columnCount())
98  m_columnCount = value.size();
99  m_data.append(value);
100  ++m_rowCount;
101  }
102 };
